<script lang="ts">
  interface Slot {
    id: string;
    time: string;
    available: boolean;
    selected: boolean;
  }

  let { slots = [], onSelect, selectedSlotId = '' }: Props = $props();
</script>

<div data-testid="slots-container" class="slots-container">
  <input type="date" data-testid="date-picker" class="date-picker" />

  <div data-testid="slot-grid" class="slot-grid">
    {#each slots as slot}
      <button
        data-testid={slot.available ? 'slot-available' : 'slot-unavailable'}
        data-testid-selected={slot.id === selectedSlotId ? 'slot-selected' : ''}
        class:selected={slot.id === selectedSlotId}
        disabled={!slot.available}
        onclick={() => onSelect?.(slot)}
      >
        {slot.time}
      </button>
    {/each}
  </div>

  <div data-testid="duration" class="duration">
    Dur√©e: 30 minutes
  </div>
  <div data-testid="time-range" class="time-range">
    08:00 - 20:00

</div>
